Card 样式说明

1.chat样式

消息格式：

{
    "type": "Chat",
    "template": "xxx",
    //asr内容（你对若琪说的话）
    "feedback": {
        "voiceUrl": null,
        "voice": "你好"
    },
    //来自于应用的 appid
    "appid": "E33FCE60E7294A61B84C43C1A171DFD8"
}

template ：

{"tts":"我是若琪，很高兴认识你"}

2.Brief summary样式

消息格式 ：

{
    "appid": "com.rokid.alarm1",
    "feedback": {
        "voice": "帮我设置一个明天早上十点的闹钟"
    },
    "id": "475b8d28-6a06-4453-993f-84c724876794",
    "template": "xxxx",
    "type": "Summary"
}

template ： 

{
    "buttons": [
        {
            "title": "查看已设置的闹钟",
            //点击跳转的url

    ],
    "icon": "https://s.rokidcdn.com/mobile-app/icon/card/alarm.png",

        {

            "subtitle": "2017年12月15日",
            "title": "10:00"
        }
    ],
    "title": "闹钟",
    "type": "Brief"
}
